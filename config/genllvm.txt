windows-script: build.bat

linux-build: pushd . && cd #genllvm# && clang #files# -lm -rdynamic -O3 -o ../#target#.exe && popd

darwin-build: pushd . && cd #genllvm# && clang #files# -lm -rdynamic -O3 -o ../#target#.exe && popd
darwin-script: build.sh

freebsd-build: pushd . && cd #genllvm# && clang #files# -lm -lexecinfo -rdynamic -O3 -o ../#target#.exe && popd
freebsd-script: build.sh
